SQLAlchemy Model (Python)
backend/models.py Python 
from sqlalchemy import Column, Integer, Text, DateTime, ForeignKey, CheckConstraint
from sqlalchemy.orm import relationship, DeclarativeBase
from datetime import datetime

class Base(DeclarativeBase): pass

class User(Base):
    __tablename__ = "users"
    id            = Column(Integer, primary_key=True, autoincrement=True)
    username      = Column(Text, unique=True, nullable=False)
    password_hash = Column(Text, nullable=False)
    email         = Column(Text, unique=True)
    created_at    = Column(DateTime, default=datetime.utcnow)
    interviews    = relationship("Interview", back_populates="user")

class Interview(Base):
    __tablename__  = "interviews"
    id             = Column(Integer, primary_key=True, autoincrement=True)
    user_id        = Column(Integer, ForeignKey("users.id"), nullable=False)
    session_uuid   = Column(Text, unique=True, nullable=False)
    job_title      = Column(Text, nullable=False)
    resume_text    = Column(Text)
    status         = Column(Text, default="ongoing")
    feedback_summary = Column(Text)
    started_at     = Column(DateTime, default=datetime.utcnow)
    ended_at       = Column(DateTime)
    user     = relationship("User", back_populates="interviews")
    messages = relationship("Message", back_populates="interview", order_by="Message.turn_index")

class Message(Base):
    __tablename__  = "messages"
    __table_args__ = (CheckConstraint("role IN ('user','assistant')"),)
    id           = Column(Integer, primary_key=True, autoincrement=True)
    interview_id = Column(Integer, ForeignKey("interviews.id"), nullable=False)
    role         = Column(Text, nullable=False)
    content      = Column(Text, nullable=False)
    turn_index   = Column(Integer, nullable=False)
    created_at   = Column(DateTime, default=datetime.utcnow)
    interview    = relationship("Interview", back_populates="messages")

class ResumeAdjustment(Base):
    __tablename__    = "resume_adjustments"
    id               = Column(Integer, primary_key=True, autoincrement=True)
    user_id          = Column(Integer, ForeignKey("users.id"), nullable=False)
    original_text    = Column(Text, nullable=False)
    adjusted_text    = Column(Text, nullable=False)
    diff_json        = Column(Text)
    keywords_added   = Column(Text)
    created_at       = Column(DateTime, default=datetime.utcnow)

SSE Streaming Chat Endpoint
backend/main.py — /chat endpoint (핵심 로직) Python 
from fastapi import APIRouter, Depends
from fastapi.responses import StreamingResponse
from openai import AsyncOpenAI
from .ai_logic import build_messages, SYSTEM_PROMPT

router = APIRouter(prefix="/api/interviews")
client = AsyncOpenAI()

@router.post("/{session_uuid}/chat")
async def chat(session_uuid: str, body: ChatRequest, db: Session = Depends(get_db)):
    interview = db.query(Interview).filter_by(session_uuid=session_uuid).first()
    history   = db.query(Message).filter_by(interview_id=interview.id).all()

    # 새 유저 메시지 DB 저장
    new_msg = Message(interview_id=interview.id, role="user",
                      content=body.user_answer, turn_index=len(history))
    db.add(new_msg); db.commit()

    # OpenAI messages 배열 구성 (전체 히스토리 포함)
    messages = build_messages(interview.resume_text, interview.job_title, history, body.user_answer)

    async def stream_tokens():
        full = ""
        async with client.chat.completions.stream(
            model="gpt-4o", messages=messages, max_tokens=600
        ) as stream:
            async for chunk in stream:
                token = chunk.choices[0].delta.content or ""
                full += token
                yield f"data: {token}\n\n"
        # 스트리밍 완료 후 AI 응답 저장
        ai_msg = Message(interview_id=interview.id, role="assistant",
                         content=full, turn_index=len(history)+1)
        db.add(ai_msg); db.commit()

    return StreamingResponse(stream_tokens(), media_type="text/event-stream")
OpenAI Prompt Engineering
SYSTEM PROMPT — 압박 면접관 페르소나
당신은 10년 경력의 날카롭고 논리적인 시니어 기술 면접관입니다. 아래 지침을 반드시 따르세요: 1. [역할] 지원자의 이력서에 근거하여 면접을 진행합니다. 직무: {job_title} / 이력서 요약: {resume_summary} 2. [꼬리물기 전략] - 지원자의 답변에서 구체성이 부족한 부분을 반드시 지적합니다. - "그래서 구체적으로 어떻게 구현했나요?", "그 결과 수치가 있나요?" 형태로 파고드세요. - 답변의 논리적 모순이나 과장된 표현을 발견하면 정중하지만 날카롭게 반박합니다. - 한 답변당 후속 질문은 반드시 1개만 생성합니다. (집중도 유지) 3. [금지 사항] - 칭찬이나 긍정적 피드백은 면접 중 하지 않습니다. - 답변을 요약해주거나 대신 설명해주지 않습니다. - 면접과 무관한 주제 전환을 절대 하지 않습니다. 4. [출력 형식] 질문만 출력합니다. 설명, 코멘트 없이 질문 문장 하나로 끝냅니다.
User Message 구조 (히스토리 포함)
backend/ai_logic.py — build_messages() Python 
def build_messages(resume_text: str, job_title: str, history: list, user_answer: str):
    """OpenAI에 전달할 messages 배열 구성 — 전체 히스토리 포함"""
    system = SYSTEM_PROMPT.format(
        job_title=job_title,
        resume_summary=resume_text[:800]  # 토큰 절감을 위해 앞 800자 사용
    )
    messages = [{"role": "system", "content": system}]

    # DB에 저장된 이전 대화 히스토리 전부 추가
    for msg in history:
        messages.append({"role": msg.role, "content": msg.content})

    # 현재 유저 답변 추가
    messages.append({"role": "user", "content": user_answer})
    return messages
